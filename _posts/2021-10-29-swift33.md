var s = "3people unFollowed me"

func solution2(_ s:String) -> String {
    var first = true
    var result = ""
    
    
    
    for c in s {
        // 공백이라면 빈칸 추가 후 첫글자(false)를 true로 하여 다음에는 첫글자임을 설정
        if c == " " {
            result += " "
            first = true
            continue
        }
        
        
        // 첫글자 first 가 true라면
        if first {
            // 문자가 숫자라면 그대로 추가
            if let test = Int(String(c)) {
                result += String(c)
            } else {
            // 문자라면 대문자 변환 후 추가
                result += String(c).uppercased()
            }
            // false로 설정
            first = false
        } else {
            // 첫 글자가 아니(false) 라면 소문자 변환 후 추가
            result += String(c).lowercased()
        }
        
    }
    
    return result
}
