###  35.프로그래머스 lv2 - 가장 큰 수 (siwft)

<br/>
<br/>


```swift

var numbers = [6,10,2]



func solution(_ numbers:[Int]) -> String {
    return ""
}

























let t = numbers.map{ String($0)}
print(t)
let tt = t.sorted{ $0 + $1 > $1 + $0 }
print(tt)

tt.joined()
tt.joined(separator: " ")



func solution2(_ numbers:[Int]) -> String {
    let temp = numbers.map { String($0) }
    let result = temp.sorted {$0 + $1 > $1 + $0}

    guard result[0] != "0" else {
        return "0"
    }

    return result.joined()
    
}

```
