### 20.프로그래머스 lv1 - 이상한 문자 만들기 (siwft)

```swift

var s = "try hello world"
























func solution2(_ s:String) -> String {
    var result : String = ""
    var count = 0
    for ch in s {
        var str = String(ch)
        
        if str == " " {
            count = 0
        } else if (count & 1) == 0 {
            str = str.uppercased()
            count += 1
        } else {
            str = str.lowercased()
            count += 1
        }
        result += String(str)
        
    }
    
    
    return result
}

solution2(s)






var ss = s.components(separatedBy: " ")
var sss = ss.map{ $0.enumerated() }
print(sss,"\n")
var ssss = ss.map{ $0.enumerated().map{$0} }
print(ssss,"\n")
var sssss = ss.map{ $0.enumerated().map {$0.offset} }
print(sssss,"\n")






func solution3(_ s:String) -> String {
    let a = s.components(separatedBy: " ").map { $0.enumerated().map { $0.offset % 2 == 0 ? $0.element.uppercased() : $0.element.lowercased() } }
    return a.map{ $0.map { $0 }.joined() }.joined(separator: " ")
}


solution3(s)
```
