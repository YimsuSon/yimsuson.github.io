### 11. 프로그래머스 lv1 - 하샤드 수 (siwft)

<br/>
<br/>


```swift


var x = 12



func solution(_ x: Int) -> Bool {
    var num = x
    var sum = 0
    
    while num > 0 {
        sum += ( num % 10 )
        num /= 10
    }
    
    return (x % sum) == 0
    
}

solution(12)



func solution2(_ x: Int) -> Bool {
    let n = Array(String(x)).map({Int(String($0))!}).reduce(0,+)
    return (x % n) == 0 ? true : false
}


solution2(x)
```
