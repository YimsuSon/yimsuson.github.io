### 7. 프로그래머스 lv1 - 행렬의 덧셈 (siwft)

<br/>
<br/>


```swift






var arr1 = [[1,2],[2,3]]
var arr2 = [[3,4],[5,6]]





arr1.count
arr1[1]
arr1[1].count
arr2[1][1]


func solution(_ arr1:[[Int]], _ arr2:[[Int]]) -> [[Int]] {
    var result = arr1
    for i in 0 ..< arr1.count {
        for j in 0 ..< arr1[i].count {
            result[i][j] += arr2[i][j]
        }
    }
    
    return result
}

func solution2(_ arr1:[[Int]], _ arr2:[[Int]]) -> [[Int]] {
    return zip(arr1, arr2).map{zip($0,$1).map{$0+$1}}
}

var zi = zip(arr1,arr2)
print(zi)



// zip(a:b)
// 튜플 쌍의 sequence 에서 a와 b의 값을 각각 묶는다
```
