```swift





var a = 5


func solution(_ num:Int) -> String {
    var a : String = " "
    if num % 2 != 0 {
        a = "Odd"
    } else {
        a = "Even"
    }
    return a
}

solution(a)




func test(_ num : Int) -> String {
    return num % 2 == 0 ? "Even" : "Odd"
}
```
