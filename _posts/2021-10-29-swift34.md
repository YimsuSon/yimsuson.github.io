var a = 2

let r = 2 % 6

// 최대공약수
func gcd2(_ a: Int,_ b : Int ) -> Int {
    
    let r = a % b
    
    if r != 0 {
        // 나머지가 0 이 아니면 b와 r을 리턴
        return gcd2(b,r)
    } else {
        // 나머지가 0 이면 b를 리턴
        return b
    }
}

gcd2(a,b)
gcd2(a,b) * b
a / gcd2(a,b) * b


// 최소공배수 =  a / 최대공약수 * b
func lcm2(_ m: Int, _ n : Int) -> Int {
    return m / gcd2(m,n) * n
}

func solution2(_ arr:[Int]) -> Int {
