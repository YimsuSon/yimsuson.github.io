func solution2(_ board:[[Int]]) -> Int {
  var checking = board
  var max = 0
  

  // 2중 반복문을 돌려 (1,1)에서부터 좌상단으로 둘러쌓인 수들 중 가장 작은 수를 더해준다
  // 열
  for i in 1..<board.count {
    // 행
    for j in 1..<board[0].count {
        
      if board[i][j] == 1 {
        
        checking[i][j] = min(checking[i-1][j-1], checking[i-1][j], checking[i][j-1]) + 1
        max = max < checking[i][j] ? checking[i][j] : max
      }
    }
  }
  
  // 행이나 열이 1개인 경우는 따로 처리해준다
  if board.count == 1 || board[0].count == 1 {
    for i in 0..<board.count {
      for j in 0..<board[0].count {
        max = max < checking[i][j] ? checking[i][j] : max
      }
    }
  }
  
  // 가장 큰 수를 제곱하여 반환해준다
  return max * max
}
