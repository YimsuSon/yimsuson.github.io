func solution(_ strings:[String], _ n:Int) -> [String] {
    let index = strings[0].index(strings[0].startIndex, offsetBy: n)

    let result = strings.sorted(by : {(s1 : String, s2 : String) -> Bool in
        if s1[index] == s2[index] {

            print(s1,s2 ,"- s1 과 s2가 같을때 ")
            return s1 < s2
        }
        else {
            
            print(s1[index] , s2[index],"- s1 과 s2가 다를때 ")
            return s1[index] < s2[index]
        }
    })
 
    return result
}


solution(strings, n )







func solution2(_ strings:[String], _ n:Int) -> [String] {
    var result = strings.sorted() // 미리 사전순으로 정렬

    // n번째 글자 기준으로 정렬
    for i in 0..<result.count {
        for j in 0..<result.count - (i + 1) {
            if Array(result[j])[n] > Array(result[j + 1])[n] {
                result.swapAt(j, j + 1)
            }
        }
    }

    return result
}


solution2(strings, n )
