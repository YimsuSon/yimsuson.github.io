<br/>


```swift


var n = 123






func solution2(_ n:Int) -> Int
{
    return  Array( String(n) ).map(  {Int ( String($0) )! } ).reduce(0,+)
    
}
solution2(n)





func solution(_ n:Int) -> Int
{
    var answer:Int = 0
    var num = n
    
    while answer > 0 {
        answer += ( num % 10)
        num /= 10
    }
    
    
    
    return answer
}
```
