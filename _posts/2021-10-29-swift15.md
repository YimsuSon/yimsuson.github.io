### 14. 프로그래머스 lv1 - 소수 찾기 (siwft)

<br/>
<br/>


```swift

var arr = [5,8,7,10]
var divisor = 5







func solution2(_ arr:[Int], _ divisor:Int) -> [Int] {
    let result = arr.filter{ ($0 % divisor) == 0 }.sorted()
    return result.count > 0 ? result : [-1]
}





func solution(_ arr:[Int], _ divisor:Int) -> [Int] {
    
    var result : [Int] = [-1]
    
    for i in arr {
        if i % divisor == 0 {
            result.append(i)
        }
    }
    
    if result.count > 1 {
        result.remove(at:0)
    }
    
    
    
    return result.sorted()
}

solution(arr,divisor)
```
