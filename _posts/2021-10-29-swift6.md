### 6. 프로그래머스 lv1 - 자연수 뒤집어 배열 만들기

<br/>
<br/>


```swift




var n:Int64 = 12345


var aa = Array(String(n))
var cc = aa.reversed()
var bb = cc.map{Int(String($0))!}
print(bb)




func solution(_ n:Int64) -> [Int] {

    return Array(String(n).reversed()).map{Int(String($0))!}

}


solution(n)



func solution2(_ n:Int64) -> [Int] {
    var num = Int(n)
    var arr = [Int]()
    
    while num > 0 {
        arr.append(num % 10)
        num /= 10
    }
    
    return arr
}


solution2(n)
```
