###  26.프로그래머스 lv1 - K번째 수 (siwft)

<br/>
<br/>


```swift

var array = [1, 5, 2, 6, 3, 7, 4]
var commands = [[2, 5, 3], [4, 4, 1], [1, 7, 3]]

// [1, 5, 2, 6, 3, 7, 4]를 2번째부터 5번째까지 자른 후 정렬합니다. [2, 3, 5, 6]의 3 번째 숫자는 5입니다.
// [1, 5, 2, 6, 3, 7, 4]를 4번째부터 4번째까지 자른 후 정렬합니다. [6]의 1 번째 숫자는 6입니다.
// [1, 5, 2, 6, 3, 7, 4]를 1번째부터 7번째까지 자릅니다. [1, 2, 3, 4, 5, 6, 7]의 3 번째 숫자는 3입니다.

let s = commands[0]
let i = s[1] - 1
























func solution(_ array:[Int], _ commands:[[Int]]) -> [Int] {
    
    var result = [Int]()
    
    for i in commands {
        let start = i[0] - 1
        let end = i[1] - 1
        
        let temp = array[start ... end].sorted()
        
        let k = i[2] - 1
        result.append(temp[k])
    }


    return result
}

solution(array,commands)




func solution2(_ array:[Int], _ commands:[[Int]]) -> [Int] {
        return commands.map({(k) in
            return array[   (k[0]-1) ... (k[1]-1)   ].sorted()   [k[2]-1]
        })
}

solution2(array,commands)
```
