### 17.프로그래머스 lv1 - 수박수박수박수 (siwft)

<br/>
<br/>


```swift

var n = 3

























func solution(_ n:Int) -> String {
    var str: String = ""

    for i in 1...n {
        if (i & 1) == 1 {
            str.append("수")
        } else {
            str.append("박")
        }
    }

    return str
}

solution(n)



func solution2(_ n:Int) -> String {
    return "\(String(repeating: "수박", count: n / 2))\(n % 2 == 0 ? "" : "수")"
}
```
