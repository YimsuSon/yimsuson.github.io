var A = [1,4,2]

func solution(_ A:[Int], _ B:[Int]) -> Int

    // [실행] 버튼을 누르면 출력 값을 볼 수 있습니다.

func solution2(_ A:[Int], _ B:[Int]) -> Int
{
    var ans = 0

    let a = A.sorted() // 오름차순
    let b = B.sorted(by: >) // 내림차순

    for i in 0..<a.count {
        ans += a[i] * b[i]
    }

    return ans
}


solution2(A,B)
