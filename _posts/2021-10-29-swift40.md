###  40.프로그래머스 lv3 - 네트워크 (siwft)

<br/>
<br/>


```swift

var n = 3
var computers = [[1,1,0],[1,1,0],[0,0,1]]

func solution(_ n:Int, _ computers:[[Int]]) -> Int {
    var visit: [Bool] = Array(repeating: false, count: n)
    var answer: Int = 0

    func bfs(_ vertax: Int) {
        visit[vertax] = true
        print(visit)
        for i in 0..<n {
            if computers[vertax][i] == 1 && visit[i] == false {
                bfs(i)
            }
        }
    }

    for i in 0..<n {
        if !visit[i] {
            answer += 1
            bfs(i)
        }
    }

    return answer
}

solution(n, computers)
```
